## 导出项目的依赖
pip freeze > requirements.txt

requests==2.22.0

pip install -r requirements.txt

请求

  • 导入相关包

    1

    import requests

  • 无参数的get请求

    1

    r=requests.get("http://www.bfreeman.cn")

  • 带参数的get请求

    1
2

    params = {'key1': 'value1', 'key2': 'value2'}
r = requests.get("http://www.bfreeman.cn", params=params)

  • post请求

    1

    r = requests.post("http://www.bfreeman.cn", data = {"key":"value"})

  • post json请求

    1
2

    params = {"key":"value"}
r = requests.post("http://www.bfreeman.cn", json = params)

  • 文件发送

    1
2

    files = {'file': open('report.xls', 'rb')}
r = requests.post("http://www.bfreeman.cn", files=files)

  • 文件发送指定额外信息

    1
2

    files = {'file': ('report.xls', open('report.xls', 'rb'), 'application/vnd.ms-excel', {'Expires': '0'})}
r = requests.post("http://www.bfreeman.cn", files=files)

  • headers

    1
2

    headers = {'content-type': 'application/json'}
r= requests.get("http://www.bfreeman.cn",headers=headers)

  • cookie

    1
2

    cookies={'cookies_are':'working'}
r = requests.get("http://www.bfreeman.cn", cookies=cookies)

  • session

    1
2
3
4

    s=requests.Session()
login_data={'form_email':'youremail@example.com','form_password':'yourpassword'}
s.post("http://www.bfreeman.cn/login",login_data)
r = s.get("http://www.bfreeman.cn")
