	    public static Date getLastMonday(DateTime dateTime) {

	                .toDate();
	    }
	
	    /**
	     * 获取上周日零点
	     *
	     * @param dateTime

	     */
	    public static Date getLastSunday(DateTime dateTime) {
	        return dateTime

	                .withTime(23, 59, 59, 0)
	                .toDate();
	    }
